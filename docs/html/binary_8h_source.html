23 #define BC_FORWARD_TO_APPLY\
24  template<class Lv, class Rv> \
25  BCINLINE auto operator () (Lv&& lv, Rv&& rv) const \
26  -> decltype(apply(lv, rv)) { \

32  BCINLINE \
