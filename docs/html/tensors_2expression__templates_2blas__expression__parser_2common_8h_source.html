126  static auto parse_expression(Stream stream, Lv left, Rv right) {
127  /*
128  * Strips transposition and scalar-multiplied from and left and right,
129  * returns a 'contents' object. --
130  *
131  * If left/right is/are transposed, calling 'dims(0), rows(), cols()'
132  * will return the non-transposed dims/rows/cols. Ergo- you should use the original parameters
133  * to access the shape of the returned value.
134  */
135  static constexpr bool lv_scalar = blas_expression_traits<Lv>::is_scalar_multiplied::value;
136  static constexpr bool rv_scalar = blas_expression_traits<Rv>::is_scalar_multiplied::value;
137 
138  using value_type = typename Lv::value_type;
139 
140  auto alpha_lv = blas_expression_traits<Lv>::get_scalar(left);
141  auto alpha_rv = blas_expression_traits<Rv>::get_scalar(right);
142  auto alpha_ = calculate_alpha<value_type, alpha_mod, lv_scalar, rv_scalar>(stream, alpha_lv, alpha_rv);
143  auto beta_ = make_constexpr_scalar<typename expression_traits<decltype(alpha_)>::allocation_tag, beta_mod, value_type>();//blas_impl::template scalar_constant<value_type, beta_mod>();
144 
145  auto left_ = greedy_evaluate(blas_expression_traits<Lv>::remove_blas_modifiers(left), stream);
146  auto right_ = greedy_evaluate(blas_expression_traits<Rv>::remove_blas_modifiers(right), stream);
147 
148  using left_t = std::decay_t<decltype(left_)>;
149  using right_t = std::decay_t<decltype(right_)>;
150  using alpha_t = std::decay_t<decltype(alpha_)>;
151  using beta_t = std::decay_t<decltype(beta_)>;
152 
153  return contents<
154  left_t,
155  right_t,
156  alpha_t,
157  beta_t,
158  blas_expression_traits<Lv>::is_transposed::value,
159  blas_expression_traits<Rv>::is_transposed::value,
160  blas_expression_traits<Lv>::is_scalar_multiplied::value,
161  blas_expression_traits<Rv>::is_scalar_multiplied::value> { left_, right_, alpha_, beta_ };
162  }
